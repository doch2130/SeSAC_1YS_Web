index.js 처음 설정
const express = require('express');
const app = express();
const port = 8080;

app.set('view engine', 'ejs');
app.use(express.urlencoded({ extended: true}));
app.use(express.json());

app.get('/', (req, res) => {
    res.render('첫 페이지');
});


app.listen(port, () => {
    console.log('server open : ', port);
});
